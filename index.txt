            function animateNumber(element, targetValue, duration = 1500, startFrom = 0) {
                const startTime = performance.now();

                function updateCount(currentTime) {
                    const elapsedTime = currentTime - startTime;
                    const progress = Math.min(elapsedTime / duration, 1);
                    const currentValue = Math.floor(startFrom + progress * (targetValue - startFrom));
                    element.textContent = currentValue;

                    if (progress < 1) {
                        requestAnimationFrame(updateCount);
                    } else {
                        element.textContent = targetValue;
                    }
                }
                requestAnimationFrame(updateCount);
            }

            // Form validation function
            function validateForm() {
                let isValid = true;

                // Name Validation
                if (contactNameInput.value.trim() === '') {
                    nameError.textContent = 'Name cannot be empty.';
                    nameError.classList.add('show');
                    contactNameInput.classList.add('invalid');
                    isValid = false;
                } else {
                    nameError.classList.remove('show');
                    contactNameInput.classList.remove('invalid');
                }

                // Email Validation
                const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
                if (contactEmailInput.value.trim() === '') {
                    emailError.textContent = 'Email cannot be empty.';
                    emailError.classList.add('show');
                    contactEmailInput.classList.add('invalid');
                    isValid = false;
                } else if (!emailRegex.test(contactEmailInput.value.trim())) {
                    emailError.textContent = 'Invalid email format.';
                    emailError.classList.add('show');
                    contactEmailInput.classList.add('invalid');
                    isValid = false;
                } else {
                    emailError.classList.remove('show');
                    contactEmailInput.classList.remove('invalid');
                }

                // Message Validation
                if (contactMessageInput.value.trim() === '') {
                    messageError.textContent = 'Message cannot be empty.';
                    messageError.classList.add('show');
                    contactMessageInput.classList.add('invalid');
                    isValid = false;
                } else if (contactMessageInput.value.trim().length < 10) {
                    messageError.textContent = 'Message is too short (minimum 10 characters).';
                    messageError.classList.add('show');
                    contactMessageInput.classList.add('invalid');
                    isValid = false;
                } else {
                    messageError.classList.remove('show');
                    contactMessageInput.classList.remove('invalid');
                }

                console.log('Form validation result:', isValid);
                return isValid;
            }

                        headers: {
                            'Content-Type': 'application/json',

                        },

                    }
